function randInt(min, max){ return Math.floor(Math.random() * (max - min + 1)) + min; }
function pick(arr){ return arr[randInt(0, arr.length - 1)]; }
function pickMany(arr, n){
  const copy = [...arr];
  const out = [];
  while(out.length < n && copy.length){
    out.push(copy.splice(randInt(0, copy.length-1), 1)[0]);
  }
  return out;
}
function clamp(v, min=0, max=100){ return Math.max(min, Math.min(max, v)); }

function generateCompany(){
  const name = `${pick(NAME_PREFIX)} ${pick(NAME_SUFFIX)}`;
  const industry = pick(INDUSTRIES);
  const revenue = randInt(80, 950);
  const factories = randInt(2, 9);
  const warehouses = randInt(3, 14);
  const suppliers = randInt(15, 120);
  const inventoryDays = randInt(12, 60);
  const leadTime = randInt(7, 45);
  const countries = pickMany(COUNTRIES, randInt(3,5));
  return { name, industry, revenue, factories, warehouses, suppliers, inventoryDays, leadTime, countries };
}

function WarRoom({onComplete}){
  const [selected, setSelected] = useState([]);
  const [confirmed, setConfirmed] = useState(false);
  const [metrics, setMetrics] = useState({cost:60, inventory:60, profit:60, delivery:60, satisfaction:60});

  function toggle(id){
    if(confirmed) return;
    setSelected(prev=>{
      if(prev.includes(id)) return prev.filter(x=>x!==id);
      if(prev.length>=3) return prev;
      return [...prev, id];
    });
  }

  function confirm(){
    if(selected.length !== 3) return;
    let m = {...metrics};
    selected.forEach(id=>{
      const action = WAR_ROOM_ACTIONS.find(a=>a.id===id);
      m.cost = clamp(m.cost + action.effect.cost);
      m.inventory = clamp(m.inventory + action.effect.inventory);
      m.profit = clamp(m.profit + action.effect.profit);
      m.delivery = clamp(m.delivery + action.effect.delivery);
      m.satisfaction = clamp(m.satisfaction + action.effect.satisfaction);
    });
    setMetrics(m);
    setConfirmed(true);
  }

  return (
    <div>
      <div className="card">
        <h2>War Room: Choose 3 Response Actions</h2>
        <p>Pick exactly three actions to respond to the crisis. Each one trades off cost,
        inventory, profit, delivery speed, or customer satisfaction — there's no perfect answer.</p>
      </div>
      <div className="card">
        {WAR_ROOM_ACTIONS.map(a=>(
          <div key={a.id}
               className={"option " + (selected.includes(a.id) ? "selected" : "") + (confirmed && !selected.includes(a.id) ? " locked":"")}
               onClick={()=>toggle(a.id)}>
            <div className="option-title">{a.title}</div>
            <div className="option-desc">{a.desc}</div>
            <div className="why-box" style={{margin:"10px 0 0"}}>{a.why}</div>
          </div>
        ))}
        <p style={{fontSize:13}}>{selected.length}/3 selected</p>
        {!confirmed && (
          <button className="btn" disabled={selected.length!==3} onClick={confirm}>
            Confirm Response Plan
          </button>
        )}
      </div>

      {confirmed && (
        <div className="card">
          <h3>Simulated Outcome</h3>
          <Bar label="Cost Control" value={metrics.cost} color="var(--teal)" suffix="%" />
          <Bar label="Inventory Health" value={metrics.inventory} color="var(--blue)" suffix="%" />
          <Bar label="Profit Impact" value={metrics.profit} color="var(--amber)" suffix="%" />
          <Bar label="Delivery Speed" value={metrics.delivery} color="var(--green)" suffix="%" />
          <Bar label="Customer Satisfaction" value={metrics.satisfaction} color="#c792ea" suffix="%" />
          <div className="btn-row">
            <button className="btn" onClick={()=>onComplete(metrics, selected)}>Proceed to Negotiation</button>
          </div>
        </div>
      )}
    </div>
  );
}

function Negotiation({onComplete}){
  const [round, setRound] = useState(0);
  const [trust, setTrust] = useState(50);
  const [price, setPrice] = useState(50);
  const [leadTime, setLeadTime] = useState(50);
  const [history, setHistory] = useState([]);
  const [choiceMade, setChoiceMade] = useState(null);

  const current = NEGOTIATION_ROUNDS[round];

  function choose(opt){
    setChoiceMade(opt);
  }

  function next(){
    const opt = choiceMade;
    const nt = clamp(trust + opt.effect.trust);
    const np = clamp(price + opt.effect.price);
    const nl = clamp(leadTime + opt.effect.leadTime);
    setTrust(nt); setPrice(np); setLeadTime(nl);
    const newHistory = [...history, {round: current.id, label: opt.label}];
    setHistory(newHistory);
    setChoiceMade(null);
    if(round < NEGOTIATION_ROUNDS.length -1){
      setRound(round+1);
    } else {
      const score = Math.round((nt + np + nl) / 3);
      onComplete({trust: nt, price: np, leadTime: nl, score, history: newHistory});
    }
  }

  return (
    <div>
      <div className="card">
        <h2>Supplier Negotiation — Round {round+1} of {NEGOTIATION_ROUNDS.length}</h2>
        <p>{current.prompt}</p>
      </div>
      <div className="card">
        {current.options.map(opt=>(
          <div key={opt.id}
               className={"option " + (choiceMade && choiceMade.id===opt.id ? "selected":"")}
               onClick={()=>choose(opt)}>
            <div className="option-title">{opt.label}</div>
            <div className="why-box" style={{margin:"10px 0 0"}}>{opt.why}</div>
          </div>
        ))}
        <div className="btn-row">
          <button className="btn" disabled={!choiceMade} onClick={next}>
            {round < NEGOTIATION_ROUNDS.length-1 ? "Next Round" : "Finish Negotiation"}
          </button>
        </div>
      </div>
      <div className="card">
        <h3>Live Negotiation Standing</h3>
        <Bar label="Trust" value={trust} color="var(--teal)" suffix="%" />
        <Bar label="Price Favorability" value={price} color="var(--amber)" suffix="%" />
        <Bar label="Lead Time Improvement" value={leadTime} color="var(--blue)" suffix="%" />
      </div>
      <div className="why-box">
        <strong>Why this matters:</strong> Negotiation isn't just about price — trust and
        lead time compound over future crises. A supplier who trusts you will bend over
        backwards next time something breaks.
      </div>
    </div>
  );
}

function Boardroom({onComplete}){
  const [qIndex, setQIndex] = useState(0);
  const [score, setScore] = useState(0);
  const [chosen, setChosen] = useState(null);
  const [answers, setAnswers] = useState([]);

  const q = CEO_QUESTIONS[qIndex];

  function choose(opt){
    setChosen(opt);
  }

  function next(){
    const newScore = score + chosen.score;
    const newAnswers = [...answers, {question: q.q, chosen: chosen.label, why: chosen.why, score: chosen.score}];
    setScore(newScore);
    setAnswers(newAnswers);
    setChosen(null);
    if(qIndex < CEO_QUESTIONS.length -1){
      setQIndex(qIndex+1);
    } else {
      const pct = Math.round((newScore / (CEO_QUESTIONS.length*10)) * 100);
      onComplete({rawScore:newScore, pct, answers:newAnswers});
    }
  }

  return (
    <div>
      <div className="card">
        <h2>CEO Boardroom — Question {qIndex+1} of {CEO_QUESTIONS.length}</h2>
        <p>{q.q}</p>
      </div>
      <div className="card">
        {q.options.map((opt,i)=>(
          <div key={i}
               className={"option " + (chosen===opt ? "selected":"")}
               onClick={()=>choose(opt)}>
            <div className="option-title">{opt.label}</div>
            {chosen===opt && <div className="why-box" style={{margin:"10px 0 0"}}>{opt.why}</div>}
          </div>
        ))}
        <div className="btn-row">
          <button className="btn" disabled={!chosen} onClick={next}>
            {qIndex < CEO_QUESTIONS.length-1 ? "Next Question" : "Finish Boardroom Session"}
          </button>
        </div>
      </div>
      <div className="why-box">
        <strong>Why this matters:</strong> Executive decisions under pressure reveal a
        company's real values. Boards and employees remember how leaders acted during a
        crisis far longer than the crisis itself.
      </div>
    </div>
  );
}

function AIStrategy({onComplete}){
  const [selected, setSelected] = useState([]);

  function toggle(id){
    setSelected(prev=>{
      if(prev.includes(id)) return prev.filter(x=>x!==id);
      if(prev.length>=2) return prev;
      return [...prev, id];
    });
  }

  function confirm(){
    const chosen = AI_INVESTMENTS.filter(a=>selected.includes(a.id));
    const totals = chosen.reduce((acc,a)=>{
      acc.resilience += a.stats.resilience;
      acc.cost += a.stats.cost;
      acc.satisfaction += a.stats.satisfaction;
      return acc;
    }, {resilience:0, cost:0, satisfaction:0});
    onComplete({chosen, totals});
  }

  return (
    <div>
      <div className="card">
        <h2>AI Strategy: Choose 2 Investments</h2>
        <p>Looking beyond this crisis — which two AI capabilities will you fund to make the
        supply chain more resilient going forward?</p>
      </div>
      <div className="card">
        {AI_INVESTMENTS.map(a=>(
          <div key={a.id}
               className={"option " + (selected.includes(a.id) ? "selected":"")}
               onClick={()=>toggle(a.id)}>
            <div className="option-title">{a.title}</div>
            <div className="option-desc">{a.desc}</div>
            <div className="tag-row">
              <span className="tag">Expected Impact</span>
            </div>
            <div className="why-box" style={{margin:"10px 0 0"}}>{a.impact}</div>
          </div>
        ))}
        <p style={{fontSize:13}}>{selected.length}/2 selected</p>
        <div className="btn-row">
          <button className="btn" disabled={selected.length!==2} onClick={confirm}>
            Lock In Strategy
          </button>
        </div>
      </div>
    </div>
  );
}

function ResultsScreen({data, onReplay}){
  const {company, crisis, warRoom, negotiation, boardroom, aiStrategy} = data;

  const resilience = clamp(Math.round(
    (warRoom.metrics.inventory * 0.3) +
    (warRoom.metrics.delivery * 0.3) +
    (aiStrategy.totals.resilience * 2) +
    10
  ));
  const costControl = clamp(Math.round((warRoom.metrics.cost * 0.6) + (negotiation.price * 0.4)));
  const riskManagement = clamp(Math.round((negotiation.trust * 0.4) + (aiStrategy.totals.resilience * 2) + (warRoom.metrics.inventory*0.2)));
  const customerSatisfaction = clamp(Math.round((warRoom.metrics.satisfaction * 0.6) + (aiStrategy.totals.satisfaction*3) + (negotiation.leadTime*0.1)));
  const leadership = clamp(boardroom.pct);
  const negotiationScore = clamp(negotiation.score);

  const overall = clamp(Math.round(
    (resilience*0.2) + (costControl*0.15) + (riskManagement*0.15) +
    (customerSatisfaction*0.15) + (leadership*0.2) + (negotiationScore*0.15)
  ));

  function grade(v){
    if(v>=85) return "Exceptional";
    if(v>=70) return "Strong";
    if(v>=55) return "Solid but Uneven";
    if(v>=40) return "Shaky";
    return "Crisis Mismanaged";
  }

  const worstBoard = [...boardroom.answers].sort((a,b)=>a.score-b.score)[0];
  const bestBoard = [...boardroom.answers].sort((a,b)=>b.score-a.score)[0];

  let biggestMistake = `Your boardroom answer on "${worstBoard.question}" cost you leadership credibility: ${worstBoard.chosen}.`;
  if(warRoom.metrics.satisfaction < 45){
    biggestMistake = "Customer satisfaction took a major hit during the War Room response — the chosen actions leaned too heavily on internal cost-saving over customer experience.";
  }

  let bestDecision = `Your call on "${bestBoard.question}" showed strong leadership judgment: ${bestBoard.chosen}.`;
  if(negotiation.trust >= 75){
    bestDecision = "Your supplier negotiation built exceptional trust — a resilient relationship that will pay off in future disruptions.";
  }

  let expertRecommendation = "Invest further in supplier risk monitoring and diversify your supplier base geographically to reduce single-point-of-failure exposure.";
  if(costControl < 50){
    expertRecommendation = "Tighten cost discipline in your crisis playbook — expedited freight and emergency sourcing are effective but unsustainable as a default response.";
  } else if(customerSatisfaction < 50){
    expertRecommendation = "Prioritize proactive customer communication protocols; satisfaction damage during a crisis often outlasts the crisis itself.";
  }

  const lessons = [
    "Crisis response is always a trade-off — there is no cost-free path through disruption.",
    "Trust built with suppliers during hard times compounds into faster, cheaper support later.",
    "Executive transparency, even with incomplete information, protects long-term credibility.",
    "AI investments in forecasting and risk monitoring reduce the frequency and severity of future crises."
  ];

  return (
    <div>
      <div className="card">
        <h2>Final Debrief: {company.name}</h2>
        <p>Crisis type: <strong style={{color:"var(--text)"}}>{crisis.title}</strong> — here's how your leadership held up under pressure.</p>
        <div className="score-ring-wrap">
          <div className="big-score" style={{color: overall>=70 ? "var(--green)" : overall>=45 ? "var(--amber)" : "var(--red)"}}>
            {overall}
          </div>
          <div>
            <div className="stat-label">Overall Crisis Score (0-100)</div>
            <div className="stat-value">{grade(overall)}</div>
          </div>
        </div>
      </div>

      <div className="grid">
        <div className="card">
          <div className="stat-label">Leadership</div>
          <Bar label="Boardroom Decisions" value={leadership} color="var(--amber)" suffix="%"/>
        </div>
        <div className="card">
          <div className="stat-label">Negotiation</div>
          <Bar label="Supplier Negotiation" value={negotiationScore} color="var(--teal)" suffix="%"/>
        </div>
        <div className="card">
          <div className="stat-label">Resilience</div>
          <Bar label="Future Readiness" value={resilience} color="var(--blue)" suffix="%"/>
        </div>
        <div className="card">
          <div className="stat-label">Cost Control</div>
          <Bar label="Financial Discipline" value={costControl} color="#c792ea" suffix="%"/>
        </div>
        <div className="card">
          <div className="stat-label">Risk Management</div>
          <Bar label="Risk Posture" value={riskManagement} color="var(--green)" suffix="%"/>
        </div>
        <div className="card">
          <div className="stat-label">Customer Satisfaction</div>
          <Bar label="Customer Experience" value={customerSatisfaction} color="#f06f9f" suffix="%"/>
        </div>
      </div>

      <div className="card">
        <h3>Biggest Mistake</h3>
        <p>{biggestMistake}</p>
        <div className="divider"></div>
        <h3>Best Decision</h3>
        <p>{bestDecision}</p>
        <div className="divider"></div>
        <h3>Expert Recommendation</h3>
        <p>{expertRecommendation}</p>
        <div className="divider"></div>
        <h3>Lessons Learned</h3>
        <ul style={{color:"var(--muted)", lineHeight:1.7}}>
          {lessons.map((l,i)=>(<li key={i}>{l}</li>))}
        </ul>
      </div>

      <div className="card">
        <h3>AI Investments Selected</h3>
        <div className="tag-row">
          {aiStrategy.chosen.map(a=>(<span className="tag" key={a.id}>{a.title}</span>))}
        </div>
      </div>

      <div className="btn-row">
        <button className="btn" onClick={onReplay}>Replay Simulation</button>
      </div>
      <div className="footer-note">Operation Lifeline — every playthrough generates a new company, crisis, and outcome.</div>
    </div>
  );
}

function App(){
  const [screen, setScreen] = useState("welcome");
  const [company, setCompany] = useState(null);
  const [crisis, setCrisis] = useState(null);
  const [warRoomResult, setWarRoomResult] = useState(null);
  const [negotiationResult, setNegotiationResult] = useState(null);
  const [boardroomResult, setBoardroomResult] = useState(null);
  const [aiResult, setAiResult] = useState(null);

  function startSimulation(){
    const c = generateCompany();
    const cr = generateCrisis(c);
    setCompany(c);
    setCrisis(cr);
    setScreen("company");
  }

  function replay(){
    setWarRoomResult(null);
    setNegotiationResult(null);
    setBoardroomResult(null);
    setAiResult(null);
    startSimulation();
  }

  const stepMap = {
    welcome: -1, company: 0, crisis: 1, warroom: 2,
    negotiation: 3, boardroom: 4, aistrategy: 5, results: 6
  };

  return (
    <div className="app">
      <div className="header">
        <div className="brand">Operation Lifeline</div>
        {screen !== "welcome" && <Stepper stepIndex={stepMap[screen]} />}
      </div>

      {screen === "welcome" && <Welcome onStart={startSimulation} />}

      {screen === "company" && company &&
        <CompanyScreen company={company} onNext={()=>setScreen("crisis")} />}

      {screen === "crisis" && crisis &&
        <CrisisScreen company={company} crisis={crisis} onNext={()=>setScreen("warroom")} />}

      {screen === "warroom" &&
        <WarRoom onComplete={(metrics, actions)=>{
          setWarRoomResult({metrics, actions});
          setScreen("negotiation");
        }} />}

      {screen === "negotiation" &&
        <Negotiation onComplete={(result)=>{
          setNegotiationResult(result);
          setScreen("boardroom");
        }} />}

      {screen === "boardroom" &&
        <Boardroom onComplete={(result)=>{
          setBoardroomResult(result);
          setScreen("aistrategy");
        }} />}

      {screen === "aistrategy" &&
        <AIStrategy onComplete={(result)=>{
          setAiResult(result);
          setScreen("results");
        }} />}

      {screen === "results" && warRoomResult && negotiationResult && boardroomResult && aiResult &&
        <ResultsScreen
          data={{
            company, crisis,
            warRoom: warRoomResult,
            negotiation: negotiationResult,
            boardroom: boardroomResult,
            aiStrategy: aiResult
          }}
          onReplay={replay}
        />}
    </div>
  );
}

const root = ReactDOM.createRoot(document.getElementById("root"));
root.render(<App />);
</script>
